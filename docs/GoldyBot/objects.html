387    def allowed_to_run(self, ctx):
388        """Checks if the command is allowed to run with currect circumstances."""
389        goldy_config = GoldyBot.config.Config(GoldyBot.files.File(GoldyBot.paths.GOLDY_CONFIG_JSON))
390        
391        # Check if guild is registered.
392        #--------------------------------
393        if str(ctx.guild.id) in goldy_config.read("allowed_guilds"):
394            guild_code_name = GoldyBot.cache.FindGuilds(goldy_config).find_object_by_id(ctx.guild.id).code_name
395            guild_config = GoldyBot.utility.guilds.config.GuildConfig(GoldyBot.config.Config(GoldyBot.files.File(GoldyBot.paths.CONFIG + f"/{guild_code_name}/config.json")))
396            
397            if guild_config.is_extenstion_allowed(self.extension_name):
398
399                if not self.required_roles_ == []:
400                    # If the required roles contain 'bot_dev' and the bot dev is running the command allow the command to execute.
401                    #----------------------------------------------------------------------------------------------------------------
402                    if "bot_dev" in self.required_roles_:
403                        if str(ctx.author.id) in GoldyBot.settings.BOT_DEVS:
404                            return True
405
406                    # Check if member has any of the required roles.
407                    #----------------------------------------------------
408                    for role_code_name in self.required_roles_:
409                        if not role_code_name == "bot_dev":
410                            role = guild_config.get_role(ctx, role_code_name)
411                            if GoldyBot.objects.member.Member(ctx).has_role(role):
412                                return True
413
414                    raise GoldyBot.errors.MemberHasNoPermsForCommand(f"The member '{ctx.author.name}' does not have the right permissions to use this command.")
415
416                else:
417                    return True
418                
419        else:
420            raise GoldyBot.errors.GuildNotRegistered(f"The guild '{ctx.guild.name}' has not been registered.")
